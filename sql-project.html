SELECT p.post_id, COUNT(l.like_id) AS likes FROM posts p LEFT JOIN likes l ON p.post_id = l.post_id GROUP BY p.post_id ORDER BY likes DESC;

4Ô∏è‚É£ Inactive Users (No Posts)
